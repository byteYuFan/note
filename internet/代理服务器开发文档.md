package main

import (
	"fmt"
	"log"
	"net/http"
	"net/http/httputil"
	"net/url"
)

func handleRequest(w http.ResponseWriter, r *http.Request) {
	// 创建目标URL
	str := "https://www.pogf.com.cn:8080" + r.URL.Path
	fmt.Println("[str]", str)
	targetURL, _ := url.Parse(str) // 替换为目标服务器的URL
	// 创建反向代理
	proxy := httputil.NewSingleHostReverseProxy(targetURL)
	data := make([]byte, 100)
	r.Body.Read(data)
	fmt.Println("[request parse]", string(data))

	// 修改请求头部
	fmt.Println("修改之前 [r.Host]:", r.Host, "[r.Method]", r.Method, "[r.URL.path]", r.URL.Path)
	r.Host = targetURL.Host
	// 转发请求到目标服务器
	r.URL.Path = "/"
	fmt.Println("修改之后 [r.Host]:", r.Host, "[r.Method]", r.Method, "[r.URL.path]", r.URL.Path)

	proxy.ServeHTTP(w, r)
}

func main() {
	// 注册请求处理函数
	http.HandleFunc("/", handleRequest)

	// 启动代理服务器并监听指定的端口
	err := http.ListenAndServe(":8090", nil)
	if err != nil {
		log.Fatal(err)
	}
}
